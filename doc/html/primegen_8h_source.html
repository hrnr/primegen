412  constexpr size_t w_64 = std::numeric_limits<uint_fast64_t>::digits;
413  constexpr size_t w_rest = w < w_64 ? 0 : (w - w_64);
414  const static double log_2 = std::log(2);
415 
416  double result;
417  double log_b;
418  uint_fast64_t high_order;
419  UIntType high_order_big;
420  std::stringstream ss;
421 
422  // compute logarithm only from highest order that fit in 64b integer, rest is truncated as effectively zero
423  high_order_big = (n >> w_rest);
424 
425  // it's sure high_order not fits to uint_64t
426  // streams are slow, but most generic conversion
427  ss << high_order_big;
428  ss >> high_order;
429 
430  log_b = std::log(high_order);
431  result = log_b + w_rest * log_2;
432
