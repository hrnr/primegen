464 template <typename UIntType, size_t w> double log(const UIntType& n) {
465  constexpr size_t w_64 = std::numeric_limits<uint_fast64_t>::digits;
466  constexpr size_t w_rest = w < w_64 ? 0 : (w - w_64);
467  const static double log_2 = std::log(2);
468 
469  double result;
470  double log_b;
471  uint_fast64_t high_order;
472  UIntType high_order_big;
473  std::stringstream ss;
474 
475  // compute logarithm only from highest order that fit in 64b integer, rest is
476  // truncated as effectively zero
477  high_order_big = (n >> w_rest);
478 
479  // it's sure high_order not fits to uint_64t
480  // streams are slow, but most generic conversion
481  ss << high_order_big;
482  ss >> high_order;
483 
484  log_b = std::log(high_order);
485  result = log_b + w_rest * log_2;
486 
487  return result;
488 }
